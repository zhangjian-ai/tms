### 1.1 技术栈

## 2. 代码规范

#### 组件 name 属性
```vue

  name: 'MainLayout',  // 使用 PascalCase
  // ...
}
</script>
```

### 2.2 Vue 3 Composition API 规范

#### setup 函数组织顺序
```vue
<script>
import { ref, computed, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'

export default {
  name: 'ComponentName',
  components: {
    // 引入的组件
  },
  props: {
    // props 定义
  },
  emits: ['event-name'],
  setup(props, { emit }) {
    // 1. 响应式数据
    const data = ref(null)
    const list = ref([])
    
    // 2. 计算属性
    const computedValue = computed(() => {
      return data.value * 2
    })
    
    // 3. 方法
    const handleClick = () => {
      // 处理逻辑
    }
    
    // 4. 生命周期钩子
    onMounted(() => {
      // 初始化逻辑
    })
    
    // 5. 返回暴露的属性和方法
    return {
      data,
      list,
      computedValue,
      handleClick
    }
  }
}
</script>
```

### 2.3 模板规范

#### 指令缩写
```vue
<!-- 推荐 -->
<div :class="className" @click="handleClick" v-if="show">
  
<!-- 不推荐 -->
<div v-bind:class="className" v-on:click="handleClick" v-if="show">
```

#### 属性顺序
```vue
<el-button
  type="primary"
  size="small"
  :disabled="loading"
  @click="handleSubmit"
>
  提交
</el-button>
```

顺序：
1. 静态属性（type, size）
2. 动态属性（:disabled, :loading）
3. 事件（@click, @change）

## 4. API 接口规范

### 4.1 Axios 实例配置
```javascript
// api/index.js
import axios from 'axios'
import config from '../config/index.js'

const api = axios.create({
  baseURL: `${config.baseURL}${config.apiPrefix}`,
  timeout: config.timeout,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
api.interceptors.request.use(
  config => {
    // 添加 token 等
    return config
  },
  error => {
    return Promise.reject(error)
  }
)

// 响应拦截器
api.interceptors.response.use(
  response => {
    return response.data
  },
  error => {
    // 统一错误处理
    return Promise.reject(error)
  }
)

export default api
```

### 4.2 API 模块化
```javascript
// api/device.js
import api from './index.js'

export const deviceApi = {
  // 获取设备列表
  getDeviceList(params) {
    return api.get('/device/list', { params })
  },
  
  // 获取设备详情
  getDeviceDetail(id) {
    return api.get(`/device/${id}`)
  },
  
  // 新增设备
  addDevice(data) {
    return api.post('/device/add', data)
  },
  
  // 更新设备
  updateDevice(id, data) {
    return api.put(`/device/${id}`, data)
  },
  
  // 删除设备
  deleteDevice(id) {
    return api.delete(`/device/${id}`)
  }
}
```

### 4.3 在组件中使用
```vue
<script>
import { ref, onMounted } from 'vue'
import { deviceApi } from '@/api/device.js'
import { ElMessage } from 'element-plus'

export default {
  setup() {
    const deviceList = ref([])
    const loading = ref(false)
    
    const fetchDeviceList = async () => {
      loading.value = true
      try {
        const res = await deviceApi.getDeviceList()
        if (res.code === 200) {
          deviceList.value = res.data
        } else {
          ElMessage.error(res.msg || '获取设备列表失败')
        }
      } catch (error) {
        ElMessage.error('网络请求失败')
        console.error(error)
      } finally {
        loading.value = false
      }
    }
    
    onMounted(() => {
      fetchDeviceList()
    })
    
    return {
      deviceList,
      loading,
      fetchDeviceList
    }
  }
}
</script>
```

## 5. 状态管理规范（Pinia）

### 5.1 Store 定义
```javascript
// stores/user.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  // State
  const userInfo = ref(null)
  const token = ref('')
  
  // Getters
  const isLoggedIn = computed(() => {
    return !!token.value
  })
  
  const username = computed(() => {
    return userInfo.value?.username || ''
  })
  
  // Actions
  function setUserInfo(info) {
    userInfo.value = info
    localStorage.setItem('userInfo', JSON.stringify(info))
  }
  
  function setToken(newToken) {
    token.value = newToken
    localStorage.setItem('token', newToken)
  }
  
  function logout() {
    userInfo.value = null
    token.value = ''
    localStorage.removeItem('userInfo')
    localStorage.removeItem('token')
  }
  
  function initUserInfo() {
    const savedUserInfo = localStorage.getItem('userInfo')
    const savedToken = localStorage.getItem('token')
    if (savedUserInfo) {
      userInfo.value = JSON.parse(savedUserInfo)
    }
    if (savedToken) {
      token.value = savedToken
    }
  }
  
  return {
    userInfo,
    token,
    isLoggedIn,
    username,
    setUserInfo,
    setToken,
    logout,
    initUserInfo
  }
})
```

### 5.2 在组件中使用
```vue
<script>
import { useUserStore } from '@/stores/user'

export default {
  setup() {
    const userStore = useUserStore()
    
    // 访问 state
    console.log(userStore.userInfo)
    
    // 访问 getters
    console.log(userStore.isLoggedIn)
    
    // 调用 actions
    userStore.setUserInfo({ username: 'admin' })
    
    return {
      userStore
    }
  }
}
</script>
```

## 13. 最佳实践

### 13.1 避免直接操作 DOM
```javascript
// 不推荐
document.getElementById('element').style.color = 'red'

// 推荐
const color = ref('red')
<div :style="{ color }">
```

### 13.2 合理使用 v-if 和 v-show
```vue
<!-- 频繁切换使用 v-show -->
<div v-show="isVisible">内容</div>

<!-- 条件很少改变使用 v-if -->
<div v-if="hasPermission">内容</div>
```

### 13.3 避免在模板中使用复杂表达式
```vue
<!-- 不推荐 -->
<div>{{ list.filter(i => i.active).map(i => i.name).join(', ') }}</div>

<!-- 推荐 -->
<div>{{ activeNames }}</div>

<script>
const activeNames = computed(() => {
  return list.value
    .filter(i => i.active)
    .map(i => i.name)
    .join(', ')
})
</script>
```

## 14. 常见问题

### 14.1 响应式丢失
```javascript
// 错误：解构会丢失响应式
const { name } = reactive({ name: 'test' })

// 正确：使用 toRefs
const state = reactive({ name: 'test' })
const { name } = toRefs(state)
```

### 14.2 ref 访问
```javascript
const count = ref(0)

// 在 setup 中需要 .value
console.log(count.value)

// 在模板中自动解包，不需要 .value
<div>{{ count }}</div>
```
